char* caesarCipher(char* s, int k) {
    int len = strlen(s);
    char* result = malloc((len + 1) * sizeof(char));
    if (!result) exit(EXIT_FAILURE); 

    k = k % 26;  

    for (int i = 0; i < len; i++) {
        char c = s[i];
        if (islower(c)) {
            result[i] = ((c - 'a' + k) % 26) + 'a';
        } else if (isupper(c)) {
            result[i] = ((c - 'A' + k) % 26) + 'A';
        } else {
            result[i] = c;
        }
    }
    result[len] = '\0';
    return result;
}
