int digitSum(long long num) {
    int sum = 0;
    while (num > 0) {
        sum += num % 10;
        num /= 10;
    }
    return sum;
}

int superDigit(char* n, int k) {
    long long sum = 0;


    for (int i = 0; n[i] != '\0'; i++) {
        sum += n[i] - '0';
    }

       sum *= k;


    while (sum >= 10) {
        sum = digitSum(sum);
    }

    return (int)sum;
}
