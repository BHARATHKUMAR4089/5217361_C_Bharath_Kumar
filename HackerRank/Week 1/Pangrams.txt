#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>
#include <string.h>


const char* pangrams(const char *s) {
    int letters[26] = {0}; 
    int count = 0;

    for (int i = 0; s[i] != '\0'; i++) {
        if (isalpha((unsigned char)s[i])) {
            char lower = tolower((unsigned char)s[i]);
            int index = lower - 'a';
            if (letters[index] == 0) {
                letters[index] = 1;
                count++;
            }
        }
    }
    return (count == 26) ? "pangram" : "not pangram";
}

int main() {
    char s[1000]; 


    if (fgets(s, sizeof(s), stdin) == NULL) {
        return 1;
    }
   
    size_t len = strlen(s);
    if (len > 0 && s[len - 1] == '\n') {
        s[len - 1] = '\0';
    }

    const char* result = pangrams(s);
    printf("%s\n", result);

    return 0;
}
