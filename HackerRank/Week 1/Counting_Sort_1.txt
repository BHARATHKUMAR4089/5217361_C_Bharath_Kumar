#include <stdio.h>
#include <stdlib.h>

#define RANGE 100  


void countingSort(int arr[], int n, int count[]) {
    for (int i = 0; i < RANGE; i++) {
        count[i] = 0;
    }
    
    for (int i = 0; i < n; i++) {
        count[arr[i]]++;
    }
}

int main() {
    int n;
    scanf("%d", &n);

    int *arr = (int *)malloc(n * sizeof(int));
    if (arr == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        return 1;
    }

    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int count[RANGE];
    countingSort(arr, n, count);

    for (int i = 0; i < RANGE; i++) {
        printf("%d", count[i]);
        if (i < RANGE - 1) {
            printf(" ");
        }
    }
    printf("\n");

    free(arr);
    return 0;
}
