#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

void miniMaxSum(int *arr, int n) {
    long total = 0;
    int min = INT_MAX;
    int max = INT_MIN;

    for (int i = 0; i < n; i++) {
        total += arr[i];
        if (arr[i] < min) min = arr[i];
        if (arr[i] > max) max = arr[i];
    }

    long minsum = total - max;
    long maxsum = total - min;

    printf("%ld %ld\n", minsum, maxsum);
}

int main() {
    int n = 5; // Hackerrank fixes N = 5 for this problem
    int *arr = (int *)malloc(n * sizeof(int));

    if (arr == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        return 1;
    }

    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    miniMaxSum(arr, n);

    free(arr);
    return 0;
}
