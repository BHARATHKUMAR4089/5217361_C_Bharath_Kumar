#include <stdio.h>
#include <string.h>

int main() {
    char a[1001], b[1001]; // buffers (assuming max length 1000, adjust if needed)

    // Read two strings from stdin
    if (scanf("%s", a) != 1) return 1;
    if (scanf("%s", b) != 1) return 1;

    int len = strlen(a);
    char result[1001];

    for (int i = 0; i < len; i++) {
        // Convert '0'/'1' to integer 0/1, XOR, then convert back to char
        result[i] = ((a[i] - '0') ^ (b[i] - '0')) + '0';
    }
    result[len] = '\0'; // null terminate

    printf("%s\n", result);

    return 0;
}
