#include <stdio.h>
#include <string.h>

int main() {
    char a[1001], b[1001];

  
    if (scanf("%s", a) != 1) return 1;
    if (scanf("%s", b) != 1) return 1;

    int len = strlen(a);
    char result[1001];

    for (int i = 0; i < len; i++) {
                result[i] = ((a[i] - '0') ^ (b[i] - '0')) + '0';
    }
    result[len] = '\0'; 

    printf("%s\n", result);

    return 0;
}
