#include <stdio.h>
#include <stdlib.h>

void plusMinus(int *arr, int n) {
    int positive = 0, negative = 0, zeroc = 0;

    for (int i = 0; i < n; i++) {
        if (arr[i] > 0) {
            positive++;
        } else if (arr[i] < 0) {
            negative++;
        } else {
            zeroc++;
        }
    }

    printf("%.6f\n", (double)positive / n);
    printf("%.6f\n", (double)negative / n);
    printf("%.6f\n", (double)zeroc / n);
}

int main() {
    int n;
    scanf("%d", &n);

    int *arr = (int *)malloc(n * sizeof(int));
    if (arr == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        return 1;
    }

    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    plusMinus(arr, n);

    free(arr);
    return 0;
}
