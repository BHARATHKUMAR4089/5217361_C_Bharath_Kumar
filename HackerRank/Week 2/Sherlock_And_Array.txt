char* balancedSums(int arr_count, int* arr) {
    long long total = 0;
    for (int i = 0; i < arr_count; i++) {
        total += arr[i];
    }

    long long left_sum = 0;
    for (int i = 0; i < arr_count; i++) {
        long long right_sum = total - left_sum - arr[i];
        if (left_sum == right_sum) {
            return "YES";
        }
        left_sum += arr[i];
    }

    return "NO";
}
